# Hello again

Hello again to a future me, and to an unknown you.
I recently logged into my Amazon Web Services account and exported the databases of a number of old blogs.

There were a number of sites hosted on node for myself, friends, (my mother).
On many of the blogs, I wrote the first entry; excited for the person that I provisioned the platform for ("your **admin** here, get ready for a great blog").
Nostalgia filled, this is me doing the same, knowing I'll one day look back at this post and feel _something_.[^gyri]

Let's find out what that feeling is, unknown reader (future Dylan you likely already know?), but I'll state my intention.
I just want to record things I find interesting, and put them out into the universe.
If the universe doesn't find my writings interesting, then I'll hopefully adapt and find more interesting things, or resign to being boring.
**But** I'm into [exploring][find me climbing, running, diving, flying], [data visualization][marimo, observable js, webgl, historical viz like du bois], [math][computation, differential equations, optimization], [science][space! did my PhD material science adjacent], [tech][compilers, functional programming, nix, ML, and passe but agents _are_ kinda cool], and might dabble in short stories; we'll see where things go.
My track record for blogging has not been the best[^laugh], here's data and a preview of the fact this site is backed by [marimo](https://marimo.io) [interactivity][my current employer!]:

```python {.marimo hide="true"}
import altair as alt
import pandas as pd

# Posting history by month, from a few sources:
#   blog       — recovered from old blogs (coffee/ghost/danny/dominica), frozen
#   observable — observablehq.com/@dmadisetti notebooks, by *created* month
#   readme.dm  — THIS blog, counted live from the post sources the RSS feed is
#                built from, so the heatmap updates itself as I publish
blog = {
    "2013-07": 4, "2013-09": 3, "2013-10": 1, "2013-11": 3,
    "2015-03": 1, "2015-06": 3, "2015-07": 1,
    "2016-03": 1, "2016-04": 1, "2016-05": 1, "2016-12": 1,
    "2019-03": 12, "2019-04": 3, "2019-06": 2, "2019-07": 3, "2019-11": 2,
    "2020-01": 1, "2021-03": 1, "2022-01": 1, "2022-03": 1,
}
observable = {
    "2021-03": 3, "2021-09": 1, "2022-02": 2, "2022-11": 2, "2022-12": 1,
    "2023-02": 2
}

def readme_dm_by_month():
    """Live count of this blog's posts by month, from the post frontmatter dates
    (the source-of-truth the RSS feed publishes). Seeds a single post if run
    outside the repo so the launch cell always shows."""
    import glob, pathlib, re
    counts: dict = {}
    for p in glob.glob("docs/posts/*.md"):
        m = re.search(r"^date:\s*(\d{4}-\d{2})",
                      pathlib.Path(p).read_text(encoding="utf-8"), re.M)
        if m:
            counts[m.group(1)] = counts.get(m.group(1), 0) + 1
    return counts or {"2026-06": 1}

monthly: dict = {}
for source in (blog, observable, readme_dm_by_month()):
    for ym, n in source.items():
        monthly[ym] = monthly.get(ym, 0) + n
LAUNCH = (2026, 2)  # readme.dm rn !

# Roll months up into quarters (compact), then fill the full year x quarter grid.
quarters = ["Q1", "Q2", "Q3", "Q4"]
agg: dict = {}
for ym, n in monthly.items():
    y, m = (int(x) for x in ym.split("-"))
    key = (y, (m - 1) // 3 + 1)
    agg[key] = agg.get(key, 0) + n
df = pd.DataFrame([
    {"year": yr, "quarter": quarters[q - 1], "posts": agg.get((yr, q), 0)}
    for yr in range(2013, 2027) for q in range(1, 5)
])

# Years across, quarters down (swapped orientation — wide and short).
ax_year = alt.X("year:O", title=None,
                axis=alt.Axis(labelAngle=0, ticks=False, domain=False, labelOverlap=False))
ax_q = alt.Y("quarter:O", sort=quarters, title=None,
             axis=alt.Axis(ticks=False, domain=False))

# GitHub-style grid: threshold buckets + a translucent-gray "empty" so it reads
# on both the light and dark themes.
grid = alt.Chart(df).mark_rect(cornerRadius=2).encode(
    x=ax_year, y=ax_q,
    color=alt.Color("posts:Q", legend=None,
        scale=alt.Scale(type="threshold", domain=[1, 3, 6],
            range=["rgba(127,127,127,0.16)", "#9be9a8", "#40c463", "#216e39"])),
    tooltip=[alt.Tooltip("year:O"), alt.Tooltip("quarter:O"),
             alt.Tooltip("posts:Q", title="posts")],
)

# Ring + label marking the launch quarter.
launch = pd.DataFrame([{"year": LAUNCH[0], "quarter": quarters[LAUNCH[1] - 1]}])
ring = alt.Chart(launch).mark_rect(
    fill=None, stroke="#c47a9c", strokeWidth=2, cornerRadius=2).encode(x=ax_year, y=ax_q)
tag = alt.Chart(launch).mark_text(
    text="readme.dm —→", align="right", dx=-10, fontSize=10,
    fontStyle="italic", color="#c47a9c").encode(x=ax_year, y=ax_q)

chart = (
    (grid + ring + tag)
    .properties(
        width=540, height=150, background="transparent",
        title=alt.TitleParams(
            "Historical personal blog posts from my teens to now",
            subtitle="courtesy of some stale AWS volume",
            anchor="start", fontSize=13, subtitleFontSize=11,
            color="#aaa", subtitleColor="#888"),
    )
    .configure_view(strokeWidth=0)
    .configure_axis(labelColor="#888", labelFontSize=11)
)
chart

```
